Replace **only** this function with:

```python
def compute_glide_range_nm(altitude_ft: float, glide_ratio: float,
                            headwind_kts: float = 0.0,
                            best_glide_kts: float = 150.0) -> float:
    """
    Compute glide range in nautical miles.

    wind_factor floor fix: the old floor of 0.4 was arbitrary and physically wrong —
    in extreme headwind conditions it falsely marked unreachable cells as reachable.
    The aerodynamically correct minimum is when headwind equals best-glide speed
    (aircraft makes zero ground progress), giving wind_factor = 0.0.
    We floor at 0.05 (5%) rather than 0.0 to avoid edge-case divide-by-zero
    downstream while still correctly marking extreme-headwind cells as unreachable.
    """
    alt_nm      = max(0, altitude_ft) / 6076.12
    base_range  = alt_nm * glide_ratio
    raw_factor  = (best_glide_kts - headwind_kts) / max(best_glide_kts, 1.0)
    wind_factor = max(0.05, raw_factor)   # physics-based floor, not arbitrary 0.4
    return round(base_range * wind_factor, 2)
```

```python
async def get_nearby_runways(lat: float, lon: float, radius_km: int = 80) -> list:
    """
    Fetch runway data for (lat, lon).  Designed to be called as a background
    asyncio.create_task() — never blocks the WebSocket tick.

    Priority:
      1. OurAirports CSV (loaded once, then instant in-memory filter)
      2. Overpass OSM   (position-cached fallback)
      3. []             — apply_runway_bonus() is a no-op
    """
    global _OA_DB, _OA_LOADING

    # ── Load OurAirports CSV on first call (guard against concurrent downloads) ─
    if _OA_DB is None and not _OA_LOADING:
        _OA_LOADING = True
        try:
            _OA_DB = await _load_ourairports()
        except Exception:
            _OA_DB = []    # treat download failure as empty → use Overpass
        finally:
            _OA_LOADING = False
    elif _OA_LOADING:
        # Another task is already downloading — skip to Overpass fallback this tick
        return await _fetch_overpass(lat, lon, radius_km)

    # ── Filter in-memory (instant) ─────────────────────────────────────────────
    if _OA_DB:
        nearby = [r for r in _OA_DB
                  if _haversine(lat, lon, r["lat"], r["lon"]) <= radius_km]
        if nearby:
            return nearby

    # ── Overpass fallback (cached) ─────────────────────────────────────────────
    return await _fetch_overpass(lat, lon, radius_km)
```

```javascript
function autoManageVoice() {
  if (!latestData) return;
  const lvl = latestData.risk?.level;

  if (lvl === "CRITICAL" || lvl === "HIGH") {
    _lowRiskTickCount   = 0;
    _voiceAutoOffWarned = false;

    // CRITICAL SAFETY RULE: even if the user manually turned voice off,
    // a CRITICAL event ALWAYS re-enables it — pilots must hear the alert.
    if (lvl === "CRITICAL" && !_voiceEnabled) {
      _voiceEnabled       = true;
      _voiceManualOverride = false;   // critical overrides manual-off
      _updateVoiceBtn();
    }

    // HIGH: only auto-enable if user has not explicitly disabled
    if (lvl === "HIGH" && !_voiceEnabled && !_voiceManualOverride) {
      _voiceEnabled = true;
      _updateVoiceBtn();
    }

  } else {
    // Risk is LOW or MODERATE
    _lowRiskTickCount++;

    // Warn at 8 ticks (~12 s) before actually disabling at 12 ticks (~18 s)
    if (_lowRiskTickCount === 8 && _voiceEnabled && !_voiceManuallySet) {
      if (!_voiceAutoOffWarned) {
        speakAlert("Risk low. Voice alerts will deactivate in 10 seconds.");
        _voiceAutoOffWarned = true;
      }
    }

    // Auto-disable after 12 consecutive safe ticks (~18 s) — only if not manually set
    if (_lowRiskTickCount > 12 && _voiceEnabled && !_voiceManuallySet) {
      _voiceEnabled       = false;
      _voiceManualOverride = false;
      _voiceAutoOffWarned  = false;
      _updateVoiceBtn();
    }
  }

  // Update degraded banner visibility
  const conf = latestData.weather?.confidence;
  const terrainLive = latestData.terrain?.elevation_live;
  const degraded = (conf === "low") || (terrainLive === false && latestData.terrain);
  const banner = document.getElementById("degradedBanner");
  if (banner) banner.style.display = degraded ? "block" : "none";
}

function _updateVoiceBtn() {
  const btn = document.getElementById("voiceBtn");
  if (!btn) return;
  btn.textContent = _voiceEnabled ? "Voice On" : "Voice Off";
  btn.classList.toggle("voice-on",  _voiceEnabled);
  btn.classList.toggle("voice-off", !_voiceEnabled);
}
```
