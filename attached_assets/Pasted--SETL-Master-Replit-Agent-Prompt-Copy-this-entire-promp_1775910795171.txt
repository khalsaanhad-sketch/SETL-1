```python
def _topsis(matrix: np.ndarray, weights: list,
            cost_cols: list | None = None) -> np.ndarray:
    """
    TOPSIS with per-criterion direction control.

    cost_cols: list of column indices where LOWER value = better (cost criteria).
               Columns NOT in cost_cols are benefit criteria (higher = better,
               but we invert them so the same min-is-best logic applies).
    Default: all columns are cost criteria (lower = safer) — preserves original
    behaviour for slope, roughness, crowd, obstacle.
    Distance is a special case: very low AND very high distance are both bad.
    We convert distance to a "distance from optimal" score before entry.
    """
    if cost_cols is None:
        cost_cols = list(range(matrix.shape[1]))

    denom = np.linalg.norm(matrix, axis=0)
    denom[denom == 0] = 1e-9
    n = matrix / denom
    v = n * np.array(weights, dtype=float)

    # For cost columns: ideal_best = min (lower = safer)
    # For benefit columns: ideal_best = max (higher = safer), ideal_worst = min
    ideal_best  = np.where(
        [i in cost_cols for i in range(v.shape[1])],
        v.min(axis=0), v.max(axis=0)
    )
    ideal_worst = np.where(
        [i in cost_cols for i in range(v.shape[1])],
        v.max(axis=0), v.min(axis=0)
    )

    d_best  = np.sqrt(((v - ideal_best)  ** 2).sum(axis=1))
    d_worst = np.sqrt(((v - ideal_worst) ** 2).sum(axis=1))

    return d_worst / (d_best + d_worst + 1e-9)
```

def apply_glide_mask(cells: list, ac_lat: float, ac_lon: float,
                     altitude_ft: float, glide_ratio: float,
                     headwind_kts: float = 0.0,
                     best_glide_kts: float = 150.0,
                     wind_speed_kts: float = 0.0,
                     wind_dir_deg: float = 0.0) -> list:
    """
    Apply glide reachability mask to each cell.

    Wind component is computed on the bearing TO each individual cell
    (not the aircraft heading), because the pilot turns to face the LZ.
    This corrects the previous over/under-estimate for off-heading cells.
    """
    R = 6371.0
    for cell in cells:
        c = cell.get("corners", [])
        if len(c) < 3:
            cell["reachable"]       = True
            cell["glide_margin_nm"] = 99.0
            continue
        clat = (c[0][0] + c[2][0]) / 2
        clon = (c[0][1] + c[2][1]) / 2

        # Bearing from aircraft to cell centre
        dlat = math.radians(clat - ac_lat)
        dlon = math.radians(clon - ac_lon)
        a    = (math.sin(dlat/2)**2 +
                math.cos(math.radians(ac_lat)) *
                math.cos(math.radians(clat)) *
                math.sin(dlon/2)**2)
        dist_nm = R * 2 * math.atan2(math.sqrt(a), math.sqrt(1-a)) / 1.852

        y = math.sin(math.radians(clon - ac_lon)) * math.cos(math.radians(clat))
        x = (math.cos(math.radians(ac_lat)) * math.sin(math.radians(clat)) -
             math.sin(math.radians(ac_lat)) * math.cos(math.radians(clat)) *
             math.cos(math.radians(clon - ac_lon)))
        bearing_to_cell = (math.degrees(math.atan2(y, x)) + 360) % 360

        # Per-cell headwind component on the glide path to that cell
        cell_headwind = compute_headwind(wind_speed_kts, wind_dir_deg, bearing_to_cell)

        max_nm = compute_glide_range_nm(altitude_ft, glide_ratio,
                                        cell_headwind, best_glide_kts)
        margin  = round(max_nm - dist_nm, 2)
        cell["reachable"]       = dist_nm <= max_nm
        cell["glide_margin_nm"] = margin
    return cells
```

def _point_in_polygon(lat: float, lon: float, coords: list) -> bool:
    """
    Ray-casting point-in-polygon test.
    NOAA coords are [lon, lat] GeoJSON order — index 0 = longitude, index 1 = latitude.
    We map: xi = longitude (compare against lon), yi = latitude (compare against lat).
    """
    n, inside = len(coords), False
    j = n - 1
    for i in range(n):
        xi, yi = coords[i][0], coords[i][1]   # xi=lon, yi=lat
        xj, yj = coords[j][0], coords[j][1]
        # Standard ray-cast: horizontal ray from point eastward
        if ((yi > lat) != (yj > lat)) and (lon < (xj - xi) * (lat - yi) / (yj - yi + 1e-9) + xi):
            inside = not inside
        j = i
    return inside
```

## STEP 7 — Create NEW file `cloud_app/services/notam_engine.py`

Create a new file at `cloud_app/services/notam_engine.py` with this content. It uses the free aviationweather.gov NOTAM API (same server as METAR and SIGMET — no API key needed):

```python
"""
SETL NOTAM Engine
Fetches active NOTAMs for airports within glide range using the free
aviationweather.gov API (same NOAA host as METAR and SIGMET).
Used to down-score runways that are NOTAMed CLOSED or contaminated.
"""
import time
import httpx

_NOTAM_URL = "https://aviationweather.gov/api/data/notam?format=json"
_NOTAM_CACHE: dict = {"key": None, "ts": 0.0, "closed": set(), "contaminated": set()}
_CACHE_TTL = 600   # 10 minutes — NOTAMs change slowly

# Keywords that indicate a runway/aerodrome is not usable for emergency landing
_CLOSED_KEYWORDS    = ("CLSD", "CLOSED", "U/S", "UNSERVICEABLE", "NOT AVBL")
_CONTAMINATED_KEYS  = ("CONTAMINATED", "SNOWBANK", "ICE", "WET SNOW", "SLUSH",
                        "NIL BRAKING", "POOR BRAKING", "FRICTION")


def _cache_key(lat: float, lon: float) -> str:
    return f"{round(lat, 1)},{round(lon, 1)}"


async def get_notam_advisories(lat: float, lon: float,
                                radius_deg: float = 0.8) -> dict:
    """
    Return sets of airport ICAO codes that are currently:
      - closed (runway/aerodrome CLSD NOTAM active)
      - contaminated (poor braking action NOTAMs active)

    Uses a bounding box of ±radius_deg (~90 km) around the aircraft.
    Cached for 10 minutes — NOTAMs are slow-changing.
    Returns {"closed": set(), "contaminated": set()} on any failure.
    """
    key = _cache_key(lat, lon)
    now = time.monotonic()

    if (_NOTAM_CACHE["key"] == key and
            now - _NOTAM_CACHE["ts"] < _CACHE_TTL):
        return {
            "closed":       _NOTAM_CACHE["closed"],
            "contaminated": _NOTAM_CACHE["contaminated"],
        }

    closed, contaminated = set(), set()
    try:
        bbox = (f"{lon - radius_deg:.3f},{lat - radius_deg:.3f},"
                f"{lon + radius_deg:.3f},{lat + radius_deg:.3f}")
        url = f"{_NOTAM_URL}&bbox={bbox}"
        async with httpx.AsyncClient(timeout=6.0) as client:
            resp = await client.get(url, headers={"User-Agent": "SETL-EFB/1.0"})
            data = resp.json()

        notams = data if isinstance(data, list) else []
        for n in notams:
            icao = (n.get("icaoLocation") or n.get("location") or "").strip().upper()
            text = (n.get("traditionalMessage") or n.get("message") or "").upper()
            if not icao or not text:
                continue
            if any(kw in text for kw in _CLOSED_KEYWORDS):
                closed.add(icao)
            if any(kw in text for kw in _CONTAMINATED_KEYS):
                contaminated.add(icao)

    except Exception:
        pass   # Non-critical — fall back gracefully to empty sets

    _NOTAM_CACHE.update({
        "key": key, "ts": now,
        "closed": closed, "contaminated": contaminated,
    })
    return {"closed": closed, "contaminated": contaminated}


def notam_runway_penalty(runway: dict, notams: dict) -> float:
    """
    Return a risk penalty for a runway based on active NOTAMs.
    runway dict must contain 'airport_ident' (ICAO code).
    Returns 0.0 (no penalty), 0.10 (contaminated), or 0.20 (closed).
    """
    if not notams:
        return 0.0
    icao = (runway.get("airport_ident") or "").strip().upper()
    if not icao:
        return 0.0
    if icao in notams.get("closed", set()):
        return 0.20   # Closed aerodrome — runway bonus fully negated
    if icao in notams.get("contaminated", set()):
        return 0.10   # Poor braking — partial penalty
    return 0.0
```

let _wsRetryDelay = 1500;   // ms — starts at 1.5s, doubles each failure, cap 30s
let _wsRetryTimer = null;
let _wsIntentionalClose = false;

function initWS() {
  if (!sessionId) return;
  _wsIntentionalClose = false;
  clearTimeout(_wsRetryTimer);

  const proto = location.protocol === "https:" ? "wss" : "ws";
  ws = new WebSocket(`${proto}://${location.host}/ws/${sessionId}`);

  ws.onopen = () => {
    _wsRetryDelay = 1500;   // reset backoff on successful connect
    document.getElementById("appNotice").textContent = "Live connection established.";
  };

  ws.onmessage = (evt) => {
    try {
      const data = JSON.parse(evt.data);
      latestData = data;
      handleWsData(data);
    } catch (e) {
      console.warn("WS parse error:", e);
    }
  };

  ws.onerror = (e) => {
    console.warn("WebSocket error:", e);
  };

  ws.onclose = (evt) => {
    if (_wsIntentionalClose) return;
    document.getElementById("appNotice").textContent =
      `Connection lost — reconnecting in ${Math.round(_wsRetryDelay / 1000)}s…`;
    _wsRetryTimer = setTimeout(() => {
      _wsRetryDelay = Math.min(_wsRetryDelay * 2, 30000);
      initWS();
    }, _wsRetryDelay);
  };
}

function closeWS() {

  clearTimeout(_wsRetryTimer);
